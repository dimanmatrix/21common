```cpp

    // –°–æ–∑–¥–∞–µ–º –≤—Ä–µ–º–µ–Ω–Ω—ã–π –∫–æ–Ω—Ç–µ–π–Ω–µ—Ä –¥–ª—è rollback

            }

    }
```

```cpp
class SetOperations {
public:
    // –ü–µ—Ä–µ—Å–µ—á–µ–Ω–∏–µ –º–Ω–æ–∂–µ—Å—Ç–≤: A ‚à© B  
    template<typename T>
    static s21::set<T> intersection(const s21::set<T>& set1, const s21::set<T>& set2) {
        s21::set<T> result;
        
        // –ü—Ä–æ—Ö–æ–¥–∏–º –ø–æ –º–µ–Ω—å—à–µ–º—É –º–Ω–æ–∂–µ—Å—Ç–≤—É –¥–ª—è —ç—Ñ—Ñ–µ–∫—Ç–∏–≤–Ω–æ—Å—Ç–∏

        const auto& larger = (set1.size() <= set2.size()) ? set2 : set1;
        
        for (const auto& element : smaller) {
            if (larger.contains(element)) {
                result.insert(element);
            }
        }
        
        return result;
    }
    
    // –†–∞–∑–Ω–æ—Å—Ç—å –º–Ω–æ–∂–µ—Å—Ç–≤: A \ B

    static s21::set<T> difference(const s21::set<T>& set1, const s21::set<T>& set2) {
        s21::set<T> result;
        
        for (const auto& element : set1) {
            if (!set2.contains(element)) {
                result.insert(element);
            }
        }
        
        return result;
    }
    
    // –°–∏–º–º–µ—Ç—Ä–∏—á–Ω–∞—è —Ä–∞–∑–Ω–æ—Å—Ç—å: (A \ B) ‚à™ (B \ A)
    template<typename T>
    static s21::set<T> symmetric_difference(const s21::set<T>& set1, const s21::set<T>& set2) {
        s21::set<T> diff1 = difference(set1, set2);
        s21::set<T> diff2 = difference(set2, set1);
        
        diff1.merge(diff2);  // –û–±—ä–µ–¥–∏–Ω—è–µ–º
        return diff1;
    }
    
    // –ü—Ä–æ–≤–µ—Ä–∫–∞ –ø–æ–¥–º–Ω–æ–∂–µ—Å—Ç–≤–∞: A ‚äÜ B
    template<typename T>

        }
        
        for (const auto& element : subset) {
            if (!superset.contains(element)) {
                return false;
            }
        }
        
        return true;
    }
};
```
